# 题目链接
数论五·欧拉函数[http://hihocoder.com/problemset/problem/1298]
# 思路
欧拉函数的性质运用，利用筛选法在O(n)的时间内求解。
# 代码
	#define MAXN 5000005
	bool isPrime[MAXN];
	int primeLst[MAXN];
	int phi[MAXN];
	int primeCnt = 0;
	void Euler(int N)
	{
		fill(isPrime, isPrime + MAXN, true);
		for (int i = 2; i <= N; i++)
		{
			if (isPrime[i])
			{
				primeLst[primeCnt++] = i;
				phi[i] = i - 1;
			}
			for (int j = 0; j < primeCnt; j++)
			{
				if (i*primeLst[j] > N)
					break;
				isPrime[i*primeLst[j]] = false;
				if (i%primeLst[j] == 0)
				{
					//φ(n*p) = φ(n) * p
					phi[i*primeLst[j]] = phi[i] * primeLst[j];
					break;
				}
				else
				{
					//φ(n*p) = φ(n) * (p-1)
					phi[i*primeLst[j]] = phi[i] * (primeLst[j]-1);
				}
			}
		}
	}
	int main() {

	#ifdef DEBUG

		//freopen("in.txt", "r", stdin);

	#endif
		int T,min=INT_MAX,ans;
		int a, b;
		cin >> a >> b;
		Euler(b);
		for (int i = b; i >=a; i--)
		{
			if (phi[i] <= min)
			{
				min = phi[i];
				ans = i;
			}
		}
		cout << ans << endl;
		return 0;
	}
