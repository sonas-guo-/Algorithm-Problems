# 题目链接
顺序三元组[http://hihocoder.com/problemset/problem/1550]

# 思路
令f[i][j]表示前i个树中以j结尾的序列有多少个，那么f[i][j]就可以从f[i-1][j]以及当前读入的数a进行转移。

# 代码
	#define MAXN 100005
	int64_t f[MAXN][4];
	int main() {
	#ifdef DEBUG
	    ifstream cin("in.txt");
		ofstream cout("out.txt");
		/*freopen("in.txt", "r", stdin);
		freopen("out.txt", "w", stdout);*/
	#endif
	    int N,a;
	    cin >> N;
	    memset(f, 0, sizeof(f));
	    for (int i = 1; i <= N; i++)
	    {
	        cin >> a;
	        f[i][1] += f[i - 1][1];
	        f[i][2] += f[i - 1][2];
	        f[i][3] += f[i - 1][3];
	        if (a-1!=0)
	            f[i][a]+=f[i - 1][a - 1];
	        else
	            f[i][a]+=1;
	    }
	    cout << f[N][3] << endl;
	    return 0;
	}
